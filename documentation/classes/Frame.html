export class Frame {

  private _id: string;
  private _nameZone: string;

  private _date: Date;


  constructor(id: string, nameZone: string, value: number, date: Date) {
    this._id = id;
    this._nameZone = nameZone;
    this._value = value;
    this._date = date;

    private _maxTemperature: number;
    private _startHour: Date;
    private _endHour: Date;
    private _actualTemperature: number;


    constructor(id: string, state: boolean, alert: string, zone: string, minTemperature: number,
                maxTemperature: number, startHour: Date, endHour: Date, actualTemperature: number) {
        this._id = id;
        this._state = state;
        this._alert = alert;
        this._zone = zone;
        this._minTemperature = minTemperature;
        this._maxTemperature = maxTemperature;
        this._startHour = startHour;
        this._endHour = endHour;
        this._actualTemperature = actualTemperature;
    }

    setZone(value: string) {

    get id(): string {

    get state(): boolean {

    get alert(): string {

    get zone(): string {

        return this._maxTemperature;

    }
}

import {Component, OnInit} from '@angular/core';
import {PsSharedState} from './../../@core/ps.shared-state';
import {Sensor} from '../../sensor';
import {PsSharedData} from '../../@core/ps.shared-data';
import {Http} from '@angular/http';
import {Zone} from '../../zone';

@Component({
    selector: 'ngx-config',
    styleUrls: ['./config.component.scss'],
    templateUrl: './config.component.html',
})
export class ConfigComponent implements OnInit {
    public zonesMap: Map<string, Zone> = new Map();
    public sensorsMap: Map<string, Sensor> = new Map();
    public sensors: Array<Sensor>;
    //
    public id = '';
    public buttonValue = false;
    public zone = '';
    public minTemp = '';
    public maxTemp = '';
    public startHour = '00:00';
    public endHour = '00:00';
    public formFieldsShowToggle = false;
    public editToggle = false;
    public alerts: Array<string> = ['GREEN', 'RED', 'ORANGE'];
    public classAlerts: Array<string> = ['btn btn-success btn-block', 'btn btn-danger btn-block', 'btn btn-warning btn-block'];
    public alertIndex = 0;
    public actualTemp = 0;


    constructor(private sharedVar: PsSharedState, private sharedSensordata: PsSharedData, private http: Http) {
    }

    ngOnInit(): void {
        this.http.get('/getSensorsData').subscribe(result => {
                const listSensors = result.json();
                listSensors.forEach((value: Sensor) => {
                    this.sensorsMap.set(value.id.replace('0x', ''), new Sensor(value.id.replace('0x', ''),
                        (Boolean)(value.state), value.alert, value.zone, value.minTemperature, value.maxTemperature,
                        new Date((value.startHour.valueOf() - 60) * 60000),
                        new Date((value.endHour.valueOf() - 60) * 60000), value.actualTemperature));
                });
                this.sensors = Array.from(this.sensorsMap.values()).sort((a: Sensor, b: Sensor) => {
                    if (a.id > b.id) {
                        return 1;
                    }
                    return -1;
                });
            },
        );
        this.http.get('/getZonesData').subscribe(result => {
                const listZones = result.json();
                listZones.forEach((value: Zone) => {
                    this.zonesMap.set(value.name, new Zone(value.name, value.department));
                });
            },
        );
    }

    initiate() {
        this.id = '';
        this.buttonValue = false;
        this.alertIndex = 0;
        this.zone = '';
        this.minTemp = '';
        this.maxTemp = '';
        this.startHour = '00:00';
        this.endHour = '00:00';
        this.formFieldsShowToggle = false;
        this.editToggle = false;
    }


    changeOnOFF(id) {
        this.sensorsMap.get(id).setState(!this.sensorsMap.get(id).state);
        const state = this.http.post('/onOffSensor',
            ['0x' + this.sensorsMap.get(id).id, this.sensorsMap.get(id).state]).subscribe();
    }


    deleteFieldValue(index) {
        if (confirm('Are you sure to delete this sensor ?')) {
            this.http.post('/deleteSensor', ['0x' + this.sensors[index].id]).subscribe();
            this.sensorsMap.delete(this.sensors[index].id);
            this.sensors.splice(index, 1);
        }
    }

    modifyFieldValue(index) {
        this.formFieldsShowToggle = true;
        this.editToggle = true;
        this.id = this.sensors[index].id;
        this.buttonValue = this.sensors[index].state;
        this.alertIndex = this.alerts.indexOf(this.sensors[index].alert);
        this.zone = this.sensors[index].zone;
        this.minTemp = this.sensors[index].minTemperature.toString();
        this.maxTemp = this.sensors[index].maxTemperature.toString();
        this.actualTemp = this.sensors[index].actualTemperature;
        let startH = '', startM = '';
        if (this.sensors[index].startHour.getHours() < 10) {
            startH = '0' + this.sensors[index].startHour.getHours().toString();
        } else {
            startH = this.sensors[index].startHour.getHours().toString();
        }
        if (this.sensors[index].startHour.getMinutes() < 10) {
            startM = '0' + this.sensors[index].startHour.getMinutes().toString();
        } else {
            startM = this.sensors[index].startHour.getMinutes().toString();
        }
        this.startHour = startH + ':' + startM;
        let endH = '', endM = '';
        if (this.sensors[index].endHour.getHours() < 10) {
            endH = '0' + this.sensors[index].endHour.getHours().toString();
        } else {
            endH = this.sensors[index].endHour.getHours().toString();
        }
        if (this.sensors[index].endHour.getMinutes() < 10) {
            endM = '0' + this.sensors[index].endHour.getMinutes().toString();
        } else {
            endM = this.sensors[index].endHour.getMinutes().toString();
        }
        this.endHour = endH + ':' + endM;
    }

    saveFieldValue() {
        if (this.id !== '' && this.zone !== '' && parseInt(this.minTemp, 10) <= 20 && parseInt(this.maxTemp, 10) >= 25) {
            if (!this.sensorsMap.has(this.id) && !this.editToggle) {
                if (!this.zonesMap.has(this.zone)) {
                    if (confirm('Zone does not exist in data base, would you like to add it ?')) {
                        this.http.post('/addZone', [this.zone, '-----']).subscribe();
                        this.zonesMap.set(this.zone, new Zone(this.zone, '-----'));
                    } else {
                        this.initiate();
                        return;
                    }
                }
                const starthour = Number(this.startHour.toString().substr(0, 2)) * 60 +
                    Number(this.startHour.toString().substr(3, 2));
                const endhour = Number(this.endHour.toString().substr(0, 2)) * 60 +
                    Number(this.endHour.toString().substr(3, 2));
                const addedSensor: Sensor = new Sensor(this.id, this.buttonValue, this.alerts[this.alertIndex],
                    this.zone, parseInt(this.minTemp, 10), parseInt(this.maxTemp, 10),
                    new Date((starthour - 60) * 60000), new Date((endhour - 60) * 60000), 0);
                this.sensorsMap.set(addedSensor.id, addedSensor);
                this.sensors.push(addedSensor);
                this.sensors.sort((a: Sensor, b: Sensor) => {
                    if (a.id > b.id) {
                        return 1;
                    }
                    return -1;
                });
                this.http.post('/addSensor',
                    ['0x' + this.id, this.buttonValue, this.alerts[this.alertIndex], this.zone, this.minTemp, this.maxTemp,
                        starthour, endhour]).subscribe();
                this.initiate();
            } else {
                if (!this.editToggle) {
                    if (confirm('this sensor already exists! Would you like to change it ?')) {
                        this.editToggle = true;
                    }
                }
                if (this.editToggle) {
                    const starthour = Number(this.startHour.toString().substr(0, 2)) * 60 +
                        Number(this.startHour.toString().substr(3, 2));
                    const endhour = Number(this.endHour.toString().substr(0, 2)) * 60 +
                        Number(this.endHour.toString().substr(3, 2));
                    const modifiedSensor: Sensor = new Sensor(this.id, this.buttonValue, this.alerts[this.alertIndex],
                        this.zone, parseInt(this.minTemp, 10), parseInt(this.maxTemp, 10),
                        new Date((starthour - 60) * 60000), new Date((endhour - 60) * 60000), 0);
                    this.sensorsMap.delete(this.id);
                    this.sensorsMap.set(this.id, modifiedSensor);
                    this.http.post('/editSensor',
                        ['0x' + this.id, this.buttonValue, this.alerts[this.alertIndex], this.zone, this.minTemp, this.maxTemp,
                            starthour, endhour]).subscribe();
                    this.sensors = Array.from(this.sensorsMap.values()).sort((a: Sensor, b: Sensor) => {
                        if (a.id > b.id) {
                            return 1;
                        }
                        if (a.id < b.id) {
                            return -1;
                        }
                        return 0;
                    });
                    this.initiate();
                }
            }
        } else {
            alert('No valid sensor to add !');
        }
    }

    showFormFields() {
        this.formFieldsShowToggle = true;
    }

    changeAlert() {
        this.alertIndex++;
        if (this.alertIndex === 3) {
            this.alertIndex = 0;
        }
    }
}

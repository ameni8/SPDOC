import {Component, OnInit} from '@angular/core';
import {Sensor} from '../../sensor';
import {PsSharedData} from '../../@core/ps.shared-data';
import {PsSharedState} from '../../@core/ps.shared-state';
import {Http} from '@angular/http';

@Component({
  selector: 'ngx-total-config',
  templateUrl: './total-config.component.html',
  styleUrls: ['./total-config.component.scss'],
})
export class TotalConfigComponent implements OnInit {

  public sensorsMap: Map<string, Sensor> = new Map();
  public minTemp = '';
  public maxTemp = '';
  public startHour = '00:00';
  public endHour = '00:00';
  public startHourNumber: number;
  public endHourNumber: number;

  ngOnInit() {
  }

  constructor(private sharedVar: PsSharedState, private sharedSensordata: PsSharedData, private httpGet: Http, private httpPost: Http) {
  }


  modifyFieldValue(index) {

  }

  initiate() {
    this.minTemp = '';
    this.maxTemp = '';
    this.startHour = '00:00';
    this.endHour = '00:00';
  }

  saveFieldValue() {
    if (parseInt(this.minTemp, 10) <= 20 && parseInt(this.maxTemp, 10) >= 25) {
      const starthour = Number(this.startHour.toString().substr(0, 2)) * 60 +
        Number(this.startHour.toString().substr(3, 2));
      const endhour = Number(this.endHour.toString().substr(0, 2)) * 60 +
        Number(this.endHour.toString().substr(3, 2));
      const addedSensor: Sensor = new Sensor('****', false, '*****',
        '*****', parseInt(this.minTemp, 10), parseInt(this.maxTemp, 10),
        new Date((starthour - 60) * 60000), new Date((endhour - 60) * 60000), 0);
      this.httpGet.post('/editTotalConfig',
        [this.minTemp, this.maxTemp,
          starthour, endhour]).subscribe();
    } else {
      alert('No valid sensor to add !');
    }
  }

}

import {Component, OnInit} from '@angular/core';
import {PsSharedData} from '../../@core/ps.shared-data';
import {Http} from '@angular/http';
import {PsSharedState} from '../../@core/ps.shared-state';
import {Sensor} from '../../sensor';
import {Observable} from 'rxjs/Observable';

@Component({
    selector: 'ngx-dashboard',
    styleUrls: ['./dashboard.component.scss'],
    templateUrl: './dashboard.component.html',
})
export class DashboardComponent implements OnInit {
    public val = this.sharedVar.buttonData.value;
    public sensorsMap: Map<string, Sensor> = new Map();
    public sensors: Array<Sensor>;
    private timer;

    ngOnInit(): void {
        this.http.get('/getSensorsData').subscribe(result => {
                const listSensors = result.json();
                listSensors.forEach((value: Sensor) => {
                    this.sensorsMap.set(value.id.replace('0x', ''), new Sensor(value.id.replace('0x', ''),
                        (Boolean)(value.state), value.alert, value.zone, value.minTemperature, value.maxTemperature,
                        new Date((value.startHour.valueOf() - 60) * 60000),
                        new Date((value.endHour.valueOf() - 60) * 60000), value.actualTemperature));
                });
                this.sensors = Array.from(this.sensorsMap.values()).sort((a: Sensor, b: Sensor) => {
                    if (a.id > b.id) {
                        return 1;
                    }
                    return -1;
                });
            },
        );
    }

    constructor(private sharedVar: PsSharedState, private sharedSensordata: PsSharedData, private http: Http) {
    }

    changeOnOFF(id) {
        this.sensorsMap.get(id).setState(!this.sensorsMap.get(id).state);
        this.sensors = Array.from(this.sensorsMap.values());
        const state = this.http.post('/onOffSensor',
            [this.sensorsMap.get(id).id, this.sensorsMap.get(id).state]).subscribe(value => {
        }, error2 => {
        });
    }
}

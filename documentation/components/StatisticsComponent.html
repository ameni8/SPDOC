import {Component, OnDestroy, OnInit} from '@angular/core';
import {NbThemeService, NbColorHelper} from '@nebular/theme';
import {Http} from '@angular/http';
import {Sensor} from '../../sensor';
import {Frame} from '../../frame';

@Component({
    selector: 'ngx-statistics',
    styleUrls: ['./statistics.component.scss'],
    templateUrl: './statistics.component.html',
})
export class StatisticsComponent implements OnDestroy, OnInit {
    public framesMap: Map<string, Array<Frame>> = new Map();
    private array: Array<string> = [];

    public date: Array<number> = [];
    public data: any;
    options: any;
    themeSubscription: any;
    public index = 0;
    public months: Array<string> = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'Sept', 'Oct', 'Nov', 'Dec'];

    constructor(private theme: NbThemeService, private http: Http) {
    }

    ngOnInit(): void {
        const months: Set<string> = new Set();
        this.http.get('/getFrames').subscribe(result => {
            const listFrames = result.json();
            listFrames.forEach((value) => {
                months.add(this.months[new Date(value[3]).getMonth()]);
                if (!this.framesMap.has(value[0])) {
                    this.framesMap.set(value[0], []);
                }
                this.framesMap.get(value[0]).push(new Frame(value[0], value[1], value[2], new Date(value[3])));
            });
            const array = Array.from(months.values()).sort(((a, b) => {
                if (this.months.indexOf(a) > this.months.indexOf(b)) {
                    return 1;
                } else {
                    return -1;
                }
            }));

            this.themeSubscription = this.theme.getJsTheme().subscribe(config => {

                const colors: any = config.variables;
                const chartjs: any = config.variables.chartjs;

                const dataBase = [];
                for (const item of Array.from(this.framesMap.keys())) {
                    const map = this.getFrameValuesById(item);
                    const tab: Array<number> = [];
                    months.forEach(month => {
                        if (!map.has(month)) {
                            tab.push(0);
                        } else {
                            tab.push(map.get(month));
                        }
                    });
                    dataBase.push({
                        data: tab,
                        label: item,
                        backgroundColor: NbColorHelper.hexToRgbA(colors.danger, 0.3),
                        borderColor: colors.danger,
                    });
                }

                this.data = {
                    labels: array,
                    datasets: dataBase,
                };

                this.options = {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        xAxes: [
                            {
                                gridLines: {
                                    display: true,
                                    color: chartjs.axisLineColor,
                                },
                                ticks: {
                                    fontColor: chartjs.textColor,
                                },
                            },
                        ],
                        yAxes: [
                            {
                                gridLines: {
                                    display: true,
                                    color: chartjs.axisLineColor,
                                },
                                ticks: {
                                    fontColor: chartjs.textColor,
                                },
                            },
                        ],
                    },
                    legend: {
                        labels: {
                            fontColor: chartjs.textColor,
                        },
                    },
                };
            });
        });
    }

    ngOnDestroy(): void {
        this.themeSubscription.unsubscribe();
    }

    getFrameValuesById(id) {
        const map: Map<string, number> = new Map();
        const map2: Map<string, number> = new Map();
        this.framesMap.get(id).forEach(frame => {
            if (map.has(this.months[frame.date.getMonth()])) {
                const val: number = parseInt(frame.value + '', 10)
                    + parseInt(map.get(this.months[frame.date.getMonth()]) + '', 10);
                map.set(this.months[frame.date.getMonth()], val);
                map2.set(this.months[frame.date.getMonth()], 1 + map2.get(this.months[frame.date.getMonth()]));
            } else {
                map.set(this.months[frame.date.getMonth()], frame.value);
                map2.set(this.months[frame.date.getMonth()], 1);
            }
        });
        this.months.forEach(month => {
            if (map.has(month)) {
                map.set(month, map.get(month) / map2.get(month));
            }
        });
        return map;
    }
}
